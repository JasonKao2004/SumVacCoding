存取上 *取值  &取地址

void swap(int *a, int *b) {
  int temp = *a;
  *a = *b;
  *b = temp;
}

 void solve() {

  int x = 3, y = 4;
  swap(&x, &y);
  cout << x << " " << y << endl;
}
